```python
s = input()
s_list = [s[i] for i in range(len(s))]

alpha = 'abcdefghijklmnopqrstuvwxyz'
a_list = [alpha[i] for i in range(len(alpha))]

result = []

for a in a_list:
    if a in s_list:
        result.append(s_list.index(a))
    else:
        result.append(-1)

for i in range(len(result)):
    print(result[i], end=' ')
```
